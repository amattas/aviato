     things that are not a plain index package: option flags, the local project

     (`*.whl`), URLs, and shell-variable tokens (`${…}`) — conservative, so a legitimate
     local/VCS install never trips it.

     references** (``foo @ git+…``) are NOT flagged: the marker fragment (a quote-bearing token)
     is ignored, and a name immediately followed by ``@`` is a direct reference, not a floating

     (``foo == 1.2.3`` is a valid exact pin), but a plain ``rest.split()`` would emit three tokens
     `['foo', '==', '1.2.3']` and the bare `foo` would be wrongly flagged as floating. Collapse the

         if skip_next:
             skip_next = False
             continue
         stripped = token.strip("'\"")

             continue

         # the `;`. This handles both the spaced (`foo>=1.0; python_version<'3.9'`, marker is a

         # genuinely floating spec glued to a marker is not silently dropped by the quote-skip.

         # carries an inner quote that survives the outer-quote strip — not a package. Skip it.

         # package (the URL/VCS ref pins it), so don't flag the bare name.

             continue

   476	class _FetchWalk:
   477	    """Walks a bashlex AST collecting what's needed to decide §11.3 fetch-execute violations."""
   478	
   479	    def __init__(self) -> None:
   480	        self.violations: list[str] = []
   481	        self.downloads: set[str] = set()  # files holding unverified fetched bytes
   482	        self.has_verifier: bool = False  # a real verify command anywhere in the block → downloads vetted
   483	        self.used: set[str] = set()  # words used by non-fetch, non-verifier commands
   484	
   485	    def walk(self, node: object) -> None:
   486	        kind = getattr(node, "kind", None)
   487	        if kind == "pipeline":
   488	            self._pipeline(node)
   489	        elif kind == "command":
   490	            self._command(node)
   491	        # GENERIC descent — every child node, whatever attribute holds it (including the inner command
   492	        # of a substitution, walked here so its own fetches are still collected). Never narrow this to
   493	        # named attributes: that is exactly how cycle-13 Group A failed open (compound/subshell/loop
   494	        # bodies and redirect-target substitutions were skipped). Whether a substitution EXECUTES is
   495	        # decided by its enclosing command in `_command`, not by the bare fact of being a substitution.
   496	        for child in _children(node):
   497	            self.walk(child)
   498	
   499	    def _pipeline(self, node: object) -> None:
   500	        elements = [p for p in node.parts if getattr(p, "kind", None) != "pipe"]
   501	        for idx, element in enumerate(elements):
   502	            # (A) a fetch's bytes flow into a downstream element. Safe only if EVERY downstream element
   503	            # is a pure-sink command — a stream cannot be checksum-verified mid-flight, so a verifier
   504	            # elsewhere never excuses it (C12-1), and a compound / subshell / wrapped non-sink counts as
   505	            # an executor (cycle-13 Group A/B).
   506	            if _subtree_has_fetch(element) and any(not _is_pure_sink(d) for d in elements[idx + 1 :]):
   507	                self.violations.append("fetch piped into a non-data-sink command")
   508	        if any(_subtree_has_fetch(e) for e in elements):
   509	            for element in elements:
   510	                if getattr(element, "kind", None) != "command":
   511	                    continue
   512	                if _redirects_to_interpreter_procsub(element):  # `curl | cat > >(bash)`
   513	                    self.violations.append("fetch written into a process substitution running a shell")
   514	                # Files written ANYWHERE in a fetch-bearing pipeline (`curl | cat > f`) hold fetched bytes.
   515	                tool = _fetch_tool(element)
   516	                self.downloads |= _fetch_output_files(element, tool) if tool else _redirect_files(element)
   517	
   518	    def _command(self, node: object) -> None:
   519	        tool = _fetch_tool(node)
   520	        if tool:
   521	            if _redirects_to_interpreter_procsub(node):
   522	                # `curl … > >(bash)` — the fetch's bytes are written straight into a shell. Execution.
   523	                self.violations.append("fetch written into a process substitution running a shell")
   524	            self.downloads |= _fetch_output_files(node, tool)
   525	            return
   526	        if _is_verifier(node):
   527	            # A real verify COMMAND (not a comment/string — the lexer dropped those) vets the block's
   528	            # DOWNLOADS. It does NOT vet a streamed fetch (handled above). Verify handling is deliberately
   529	            # BLOCK-LEVEL, not order/gating/artifact aware — an order-aware attempt (cycle-13 R5) was
   530	            # reverted (cycle-14 R1) because it false-positived every legitimate verified install whose
   531	            # checksum names the artifact (`printf "%s  %s" "$SHA" tool | sha256sum -c -`) or whose setup
   532	            # touches it (`chmod +x tool`) before the verifier ran. The cost: a verify that runs AFTER
   533	            # the use, fails open (`|| true`), or covers a DIFFERENT artifact still vets — documented
   534	            # out-of-scope for this honest-mistake gate (precise verify dataflow is the irreducible tail).
   535	            self.has_verifier = True
   536	            return
   537	        # (B) a fetch substitution that is EXECUTED — run as the command (`$(curl) …`) or fed to an
   538	        # interpreter (`bash -c "$(curl)"`, `bash <(curl)`, `bash <<<"$(curl)"`). A substitution merely
   539	        # CAPTURED as data (`X=$(curl)`, `echo "$(curl)"`) is not flagged — that is what keeps the gate
   540	        # usable on the dominant "fetch a value" pattern.
   541	        if _command_position_fetch_subst(node):
   542	            self.violations.append("fetch substitution run as a command")
   543	        elif _interpreter_executes_fetch(node):
   544	            self.violations.append("fetch substitution executed by an interpreter")
   545	        self.used |= set(_plain_words(node)) | _input_redirect_files(node)  # `bash < f` USES f
