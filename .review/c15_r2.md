   194	def _chain(load, name: str) -> list:
   195	    """Walk an ``extends`` chain from ``name`` to its root, root-first.
   196	
   197	    ``load`` maps a bundle name to its dataclass. Ancestors resolve before
   198	    descendants (§4.2 deterministic ordering).

   202	    current: str | None = name

   205	            raise CompositionError(f"cyclic extends chain at {current!r}")

   214	def _resolve_list(layers: list, base_attr: str) -> tuple[str, ...]:
   215	    root = layers[0]
   216	    if root.extends is not None:  # defensive; root by construction has no extends
   217	        raise CompositionError(f"resolution root {root.name!r} unexpectedly extends another")
   218	    if root.add or root.remove:
   219	        # add/remove are relative to a base layer; the root IS the base, so they have no
   220	        # meaning here. Silently ignoring them would let a misplaced override no-op (§4.2).
   221	        raise CompositionError(
   222	            f"resolution root {root.name!r} declares add/remove, which only apply under extends (§4.2)"
   223	        )
   224	    resolved = list(getattr(root, base_attr))
   225	    for layer in layers[1:]:
   226	        if getattr(layer, base_attr):
   227	            raise CompositionError(
   228	                f"bundle {layer.name!r} restates a bare {base_attr} list under extends; use add/remove (§4.2)"
   229	            )
   230	        resolved = merge_list(resolved, add=layer.add, remove=layer.remove)
   231	    return tuple(resolved)
